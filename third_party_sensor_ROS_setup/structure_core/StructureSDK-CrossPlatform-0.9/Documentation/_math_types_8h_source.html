430 inline Matrix4 Matrix4::lookAt(const Vector3f& eyePosition, const Vector3f& lookAt, const Vector3f& up)
431 {
432  Matrix4 m;
433 
434  Vector3f fwd = (lookAt - eyePosition).normalized();
435  Vector3f nUp = up.normalized();
436  Vector3f right = fwd.cross(nUp).normalized();
437 
438  nUp = right.cross(fwd);
439 
440  m.m[0] = right.x;
441  m.m[1] = nUp.x;
442  m.m[2] = -fwd.x;
443  m.m[3] = 0.0f;
444 
445  m.m[4] = right.y;
446  m.m[5] = nUp.y;
447  m.m[6] = -fwd.y;
448  m.m[7] = 0.0f;
449 
450  m.m[8] = right.z;
451  m.m[9] = nUp.z;
452  m.m[10] = -fwd.z;
453  m.m[11] = 0.0f;
454 
455  m.m[12] = -right.dot(eyePosition);
456  m.m[13] = -nUp.dot(eyePosition);
457  m.m[14] = fwd.dot(eyePosition);
458  m.m[15] = 1.0f;
459 
460  return m;
461 }

463 inline Matrix4 Matrix4::projection(float fovY, float aspectRatio, float minZ, float maxZ, Handedness handedness)
464 {
465  Matrix4 m;
466 
467  const float t = minZ * std::tan(fovY * 0.5f);
468  const float r = t * aspectRatio;
469 
470  // for right handed coordinate system
471  m.m[0] = minZ / r;
472  m.m[1] = 0.0f;
473  m.m[2] = 0.0f;
474  m.m[3] = 0.0f;
475 
476  m.m[4] = 0.0f;
477  m.m[5] = minZ / t;
478  m.m[6] = 0.0f;
479  m.m[7] = 0.0f;
480 
481  m.m[8] = 0.0f;
482  m.m[9] = 0.0f;
483  m.m[10] = -(maxZ + minZ) / (maxZ - minZ);
484  m.m[11] = -1.0f;
485 
486  m.m[12] = 0.0f;
487  m.m[13] = 0.0f;
488  m.m[14] = -2.0f * maxZ * minZ / (maxZ - minZ);
489  m.m[15] = 0.0f;
490 
491  if (handedness == Handedness::Left)
492  {
493  m.m[11] = 1.0f;
494  m.m[14] *= -1.0f;
495  }
496 
497  return m;
498 }

516 inline Matrix4 Matrix4::operator*(const Matrix4& rv) const
517 {
518  Matrix4 m_;
519 
520  m_.m00 = m00 * rv.m00 + m10 * rv.m01 + m20 * rv.m02 + m30 * rv.m03;
521  m_.m10 = m00 * rv.m10 + m10 * rv.m11 + m20 * rv.m12 + m30 * rv.m13;
522  m_.m20 = m00 * rv.m20 + m10 * rv.m21 + m20 * rv.m22 + m30 * rv.m23;
523  m_.m30 = m00 * rv.m30 + m10 * rv.m31 + m20 * rv.m32 + m30 * rv.m33;
524 
525  m_.m01 = m01 * rv.m00 + m11 * rv.m01 + m21 * rv.m02 + m31 * rv.m03;
526  m_.m11 = m01 * rv.m10 + m11 * rv.m11 + m21 * rv.m12 + m31 * rv.m13;
527  m_.m21 = m01 * rv.m20 + m11 * rv.m21 + m21 * rv.m22 + m31 * rv.m23;
528  m_.m31 = m01 * rv.m30 + m11 * rv.m31 + m21 * rv.m32 + m31 * rv.m33;
529 
530  m_.m02 = m02 * rv.m00 + m12 * rv.m01 + m22 * rv.m02 + m32 * rv.m03;
531  m_.m12 = m02 * rv.m10 + m12 * rv.m11 + m22 * rv.m12 + m32 * rv.m13;
532  m_.m22 = m02 * rv.m20 + m12 * rv.m21 + m22 * rv.m22 + m32 * rv.m23;
533  m_.m32 = m02 * rv.m30 + m12 * rv.m31 + m22 * rv.m32 + m32 * rv.m33;
534 
535  m_.m03 = m03 * rv.m00 + m13 * rv.m01 + m23 * rv.m02 + m33 * rv.m03;
536  m_.m13 = m03 * rv.m10 + m13 * rv.m11 + m23 * rv.m12 + m33 * rv.m13;
537  m_.m23 = m03 * rv.m20 + m13 * rv.m21 + m23 * rv.m22 + m33 * rv.m23;
538  m_.m33 = m03 * rv.m30 + m13 * rv.m31 + m23 * rv.m32 + m33 * rv.m33;
539 
540  return m_;
541 }

742  // Heading
743  double h = atan2(R.atRowCol(1, 0), R.atRowCol(0, 0));
744 
745  // Compute cos & sin of heading
746  double ch = cos(h);
747  double sh = sin(h);
748 
749  //Pitch
750  double p = atan2(-R.atRowCol(2, 0), R.atRowCol(0, 0)*ch + R.atRowCol(1, 0)*sh);
751 
752  // Roll
753  double r = atan2(R.atRowCol(0, 2)*sh - R.atRowCol(1, 2)*ch, -R.atRowCol(0, 1)*sh + R.atRowCol(1, 1)*ch);
754 
755  return Vector3f(float(r), float(p), float(h));
756 }
